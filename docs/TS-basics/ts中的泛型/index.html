TS 中的泛型

泛型的基本使用​

认识泛型,就是广泛的类型.

参数的类型的是固定的,不够灵活

function foo(name: string, grade: number) {}
foo('qwe', 1);
foo('ewq', 2);
  • 解决方法 1 any
function foo(arg) {
  return arg;
}

let res0 = foo(123); // any
let res1 = foo(`123`); // any
let res2 = foo({ name: 'qwe' }); // any
  • 解决方法 2 联合类型
function foo(arg: number | string | { name: string }) {
  return arg;
}

let res0 = foo(123); // number | string | { name: string };
let res1 = foo(`123`); // number | string | { name: string };
let res2 = foo({ name: 'qwe' }); // number | string | { name: string };
  • 解决方法 3 函数重载
function foo(arg: number): number;
function foo(arg: string): string;
function foo(arg: { name: string }): { name: string };
function foo(arg: any) {
  return arg;
}


let res0 = foo(123); // number
let res1 = foo(`123`); // string
let res2 = foo({ name: 'qwe' }); // { name: 'qwe' }

函数的参数化​

上面的方法都比较复杂

使用泛型, 把类型参数化

function foo<Type>(arg: Type): Type {
  return arg;
}

let res0 = foo < number > 123; // number
let res1 = foo < string > `123`; // string
let res2 = foo < { name: string } > { name: 'qwe' }; // { name: 'qwe' }

省略写法

TS 能根据参数推导出类型

let res1 = foo('qqq'); // string
let res2 = foo(111); // number

useState​

function useState<Type>(initialState: Type): [Type, (newState: Type) => void] {
  let state = initialState;
  function setState(newState: Type) {
    state = newState;
  }

  return [state, setState];
}

const [count, setCount] = useState(100); // number , (newState:number)=>void

const [message, setMessage] = useState('qwe'); // string , (newState:string)=>void

const [info, setInfo] = useState([1, 2, 3]); // number[] , (newState:number[])=>void

const [arr, setArr] = useState([]); // never[] , (newState:never[])=>void

const [arr, setArr] = useState<any[]>([]); // any[] , (newState:any[])=>void

泛型内传入多个类型​

function foo<Type, Element>(arg1: Type, arg2: Element) {
  return {
    arg1,
    arg2,
  };
}

let res = foo(1, 'qwe');
foo(true, { name: 'qwe' });

泛型接口的使用​

interface IKun<T> {
  name: string;
  grade: T;
  slogan: string;
}

const kun: IKun<number | string> = {
  name: '',
  grade: 1,
  slogan: '',
};

泛型接口的默认值

interface IKunDefault<T = number> {
  name: string;
  grade: T;
  slogan: string;
}

const kun1: IKunDefault = {
  name: '',
  grade: 1,
  slogan: '',
};

泛型约束​

interface ILength {
  length: number;
}

function getLength(arg: ILength) {
  return arg.length;
}

let res1 = getLength('123'); // number
let res2 = getLength([1, 1, 1]); // number[]
let res3 = getLength({ length: 100 }); // {length:number}

类型丢失

function getInfo<T>(arg: T) {
  return arg;
}
let res1 = getInfo('123'); // ILength
let res2 = getInfo([1, 1, 1]); // number[]
let res3 = getInfo({ length: 100 }); // {name:number}
let res4 = getInfo(Symbol('qwe')); // symbol 没有ILength的属性,不符合我们的编码需求,但是会通过编译

泛型约束

function getInfo<T extends ILength>(arg: T) {
  return arg;
}
let res1 = getInfo('123'); // ILength
let res2 = getInfo([1, 1, 1]); // number[]
let res3 = getInfo({ length: 100 }); // {name:number}
let res4 = getInfo(Symbol('qwe')); // error 类型“symbol”的参数不能赋给类型“ILength”的参数。ts(2345)

<T extends ILength> 表示的是传入的参数的类型必须有这个属性,也可以有其他的属性,但是至少要有这个成员

使用泛型约束​

我们希望获取一个对象给定属性名的值

  • 我们需要确保我们不会获取 obj 上不存在的属性
  • 所以我们在两个类型之间建立一个约束

传入的 key 类型,是 object 当中 key 的其中之一

function getObjectProperty<O, K extends keyof O>(object: O, key: K) {
  return object[key];
}

// ===

// function getObjectProperty<O, K extends 'name'|'grade'|'qqq'|'
// www'>(object: O, key: K) {
//   return object[key];
// }


const info = {
  name: 'qwe',
  grade: 1,
  qqq: 'q',
  www: 'w',
};

const name = getObjectProperty(info, 'name');
getObjectProperty(info, 'address');
// 类型“"address"”的参数不能赋给类型“"name" | "grade" | "qqq" | "www"”的参数

keyof​

interface IKun {
  name: string;
  grade: number;
}

type IKunKeys = keyof IKun; // 'name' | 'grade'

映射类型​

有时候,一个类型需要基于另外一个类型,但是你又不想拷贝一份,这个时候你可以考虑使用映射类型

映射类型是建立在索引签名上的语法

  • 映射类型, 就是使用了 PropertyKeys联合类型的泛型
  • 其中 PropertyKeys 多是通过 keyof创建,然后循环遍历键名创建一个类型
type MapPerson<T> = {
  // 索引类型依次循环使用一次
  [key_ in keyof T]: T[key_];
  /**
   * name: string
   * grade: number
   */
};

interface IPerson {
  readonly name: string;
  grade?: number;
}

type CopyPerson = MapPerson<IPerson>;

/**
  type CopyPerson = {
    name: string;
    grade: number;
  }
*/
type MapPerson<T> = {
  readonly [Property in keyof T]?: T[Property];
};

type CopyPerson = MapPerson<IPerson>;

/**
  type CopyPerson = {
    readonly name?: string | undefined;
    readonly grade?: number | undefined;
    readonly address?: string | undefined;
  }
 */

修饰符前可以加符号​

添加一个 - 号, 对被映射的类型属性修饰符进行删剪

type MapPerson<T> = {
  -readonly [Property in keyof T]-?: T[Property];
};

type CopyPerson1 = MapPerson<CopyPerson>;
/**
  type CopyPerson1 = {
    name: string
    grade: number
    address: string
  }
 */
